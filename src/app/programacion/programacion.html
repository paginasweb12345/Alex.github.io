numeros = [1, 2, 3, 4, 5, 6]
pares = [n for n in numeros if n % 2 == 0]

Ejemplo de Ordenamiento

def burbuja(lista):
    for i in range(len(lista)):
        for j in range(len(lista)-i-1):
            if lista[j] > lista[j+1]:
                lista[j], lista[j+1] = lista[j+1], lista[j]
    return lista

print(burbuja([5, 2, 9, 1]))  # [1, 2, 5, 9]
